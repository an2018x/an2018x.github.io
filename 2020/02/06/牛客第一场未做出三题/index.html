#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <string>
using namespace std;
const int maxn=100010l;
vector<int> mp[maxn];
vector<int> v;
long long val[maxn];
string str;
long long cnt;
void dfs(int i){
    val[i]=1;
    v.push_back(i);
    for(int j=0;j<mp[i].size();j++){
        if(str[mp[i][j]]=='W'&&!val[mp[i][j]]){
            cnt++;
            dfs(mp[i][j]);
        }
    }
}
int main(){
    ios::sync_with_stdio(false);
    int n;
    int a,b;
    cin>>n;
    int m=n;
    cin>>str;
    for(int i=0;i<n-1;i++){
        cin>>a>>b;
        a--;b--;
        mp[a].push_back(b);
        mp[b].push_back(a);
    }
    for(int i=0;i<m;i++){

        if(str[i]=='W'&&!val[i]){
            v.clear();
            cnt=1;
            dfs(i);
            for(auto i:v){
                val[i]=cnt;
            }
        }

    }
    long long sum=0;
    long long ans=0;
    for(int i=0;i<m;i++){
        ans=0;
        if(str[i]=='B'){
            for(int j=0;j<mp[i].size();j++){
                if(str[mp[i][j]]=='W'){
                    sum+=ans*val[mp[i][j]]+val[mp[i][j]];
                    ans+=val[mp[i][j]];
                }
            }
        }
    }
    printf("%lld\n",sum);
    return 0;
}
