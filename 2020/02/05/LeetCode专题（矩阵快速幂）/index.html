LeetCode专题（矩阵快速幂）

70.爬楼梯

题目链接 https://leetcode-cn.com/problems/climbing-stairs/submissions/

class Solution {
public:
    vector<vector<long long >> mul(const vector<vector<long long>> &A,const vector<vector<long long>> &B){
        vector<vector<long long>> C(A.size(),vector<long long>(B[0].size()));
        for(int i=0;i<A.size();i++){
            for(int k=0;k<B.size();k++){
                for(int j=0;j<B[0].size();j++){
                    C[i][j]=C[i][j]+A[i][k]*B[k][j];
                }
            }
        }
        return C;
    }
    int mutiply(vector<vector<long long>> A,int n){
        vector<vector<long long>> B(A.size(),vector<long long>(A.size()));
        for(int i=0;i<B.size();i++){
            B[i][i]=1;
        }
        while(n>0){
            if(n&1) B=mul(B,A);
            A=mul(A,A);
            n>>=1;
        }
        return B[0][0];
    }
    int climbStairs(int n) {
        vector<vector<long long>> mat(2,vector<long long>(2));
        mat[0][0]=1;
        mat[0][1]=1;
        mat[1][0]=1;
        mat[1][1]=0;
        return mutiply(mat,n);
    }
};
