A.格点三角形

一道标准的模拟题，在纸上画一下就可以退出基本的公式，要注意减去重复的部分，就是和x轴y轴都平行的部分，思路很清晰，结果我一连wa了四发，最后发现居然还有个100 100的输入样例，微调一下，减去重复的部分，就过了，还有就是要注意取模的几个基本公式。

#include <cstdio>
#include <cmath>
using namespace std;
const long long mod = 1000000007;
int main(){
    long long n,m;
    scanf("%lld%lld",&n,&m);
    n--;
    m--;
    long long res=0;
    res+=(((((n+1)%mod)*(n%mod)*2)%mod)*(m-1))%mod;
    res+=(((((m+1)%mod)*(m%mod)*2)%mod)*(n-1))%mod;
    res+=(((((m-1)%mod)*((m+1)%mod)*2)%mod)*(n))%mod;
    res+=(((((n-1)%mod)*((n+1)%mod)*2)%mod)*(m))%mod;
    // res=(res%mod-((m/2*n+n/2*m)*4)%mod+mod)%mod;
    res=res%mod-((m-1)*n+(n-1)*m)*4%mod+mod;
    printf("%lld\n",res%mod);
    return 0;

}

E.rin和快速迭代

不知道有没有什么数论方面的奇淫巧计，简单估计了复杂度，暴力模拟也能过，就是算当前数因子个数，然后再算因子个数的个数。。。。知道因子数为，注意要开long long的数据范围。

#include <iostream>
#include <cmath>
using namespace std;
long long num(long long n){ 
    long long count=2;
    for(long long i=2;i<=sqrt(n);i++){
        if(n%i==0){
            if(i==sqrt(n) && n/i==i){ 
                 count++;   
            }
            else count+=2;
        }
    }
    return count;
}
int main(){
    long long n;
    cin>>n;
    long long cnt=1;
    n=num(n);
    while(n!=2){
        cnt++;
        n=num(n);   
    }
    cout<<cnt<<endl;
    return 0;
}

I.nico和niconiconi

做这题目我满脑子都是niconiconi，其实就是个简单的线性dp问题，dp[i]，表示已当前第i个字符结尾的字符串的得分，然后就是切割字符串来匹配了。

#include <cstdio>
#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
const int maxn=300010;
long long dp[maxn];
string str;
int main(){
    long long  n,a,b,c;
    scanf("%lld%lld%lld%lld",&n,&a,&b,&c);
    cin>>str;
    long long res=0;
    for(int i=3;i<n;i++){
        dp[i]=res;
        if(str.substr(i-3,4)=="nico"){

            dp[i]=max(dp[i],dp[i-3]+a);

        }
        if(i>=5&&str.substr(i-5,6)=="niconi")
            dp[i]=max(dp[i],dp[i-5]+b);
        if(i>=9&&str.substr(i-9,10)=="niconiconi")
            dp[i]=max(dp[i],dp[i-9]+c);
        res=dp[i];
    }
    printf("%lld",dp[n-1]);
    return 0;
}
