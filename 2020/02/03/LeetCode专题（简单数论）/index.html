268.缺失数字

题目链接 https://leetcode-cn.com/problems/missing-number/

class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int sum=0;
        int n=nums.size();
        for(auto num:nums){
            sum+=num;
        }
        return n*(n+1)/2-sum;
    }
};

完整的和减去当前的和就可以了。

462.最少移动次数使数组元素相等

题目链接 https://leetcode-cn.com/problems/minimum-moves-to-equal-array-elements-ii

class Solution {
public:
    int minMoves2(vector<int>& nums) {
        int n=nums.size();
        nth_element(nums.begin(),nums.begin()+n/2,nums.end());
        int x=nums[n/2];
        int res=0;
        for(auto num:nums){
            res+=abs(x-num);
        }
        return res;
    }
};

458.可怜的小猪

题目链接 https://leetcode-cn.com/problems/poor-pigs/

class Solution {
public:
    int poorPigs(int b, int m, int p) {
        if(b==1&&m==1) return 0;
        int s=p/m+1;
        int res=s;
        int c=1;
        while(res<b) {res*=s; c++;}
        return c;
    }
};

650.只有两个数的键盘

题目链接 https://leetcode-cn.com/problems/2-keys-keyboard/submissions/

class Solution {
public:
    int minSteps(int n) {
        if(n==1) return 0;
        for(int i=n/2;i>=2;i--){
            if(!(n%i)){
                return minSteps(i)+n/i;
            }
        }
        return n;
    }
};

就是寻找最大合数，如果是质数的话，就直接返回n。
